# Rweb3.js - Web3.js for Runebase

Rweb3 is a library for dApps to interract with the Runebase blockchain. Rweb3 communicates to a Runebase node via the provider provided.

https://www.npmjs.com/package/rweb3

## Get Started
Run the following in your project folder:

	npm install rweb3 --save

## Rweb3.js
Instantiate a new instance of `Rweb3`: 
```
const { Rweb3 } = require('rweb3');

// Instantiate Rweb3 with HttpProvider
// Pass in the URL of your Runebase node RPC port with auth credentials.
// Default Runebase RPC ports: testnet=13889 mainnet=3889
const rweb3 = new Rweb3('http://bodhi:bodhi@localhost:13889');

// Instantiate Rweb3 with QryptoRPCProvider
// QryptoRPCProvider is a provider for the Qrypto Wallet Chrome Extension.
// Please note QryptoRPCProvider only allows the rawCall() method to be used.
// It is specifically used for sendtocontract and callcontract only.
const rweb3 = new Rweb3(new QryptoRPCProvider());
```

### isConnected()
Checks if you are connected properly to the local runebase node.
```
async function isConnected() {
  return await rweb3.isConnected();
}
```

### getHexAddress(address)
Converts a Runebase address to hex format.
```
async function getHexAddress() {
  return await rweb3.getHexAddress('qKjn4fStBaAtwGiwueJf9qFxgpbAvf1xAy');
}
```

### fromHexAddress(hexAddress)
Converts a hex address to Runebase format.
```
async function fromHexAddress() {
  return await rweb3.fromHexAddress('17e7888aa7412a735f336d2f6d784caefabb6fa3');
}
```

### getBlockCount()
Gets the current block height of your local Runebase node.
```
async function getBlockCount() {
  return await rweb3.getBlockCount();
}
```

### getTransaction(txid)
Gets the transaction details of the transaction id.
```
async function getTransaction(args) {
  const {
    transactionId, // string
  } = args;

  return await rweb3.getTransactionReceipt(transactionId);
}
```

### getTransactionReceipt(txid)
Gets the transaction receipt of the transaction id.
```
async function getTransactionReceipt(args) {
  const {
    transactionId, // string
  } = args;

  return await rweb3.getTransactionReceipt(transactionId);
}
```

### listUnspent()
Gets the unspent outputs that can be used.
```
async function listUnspent() {
  return await rweb3.listUnspent();
}
```

Usage:
```
const ContractMetadata = require('./contract_metadata');

async function(args) {
  let {
    fromBlock, // number
    toBlock, // number
    addresses, // string array
    topics // string array
  } = args;

  if (addresses === undefined) {
    addresses = [];
  }
  if (topics === undefined) {
    topics = [];
  }

  // removeHexPrefix = true removes the '0x' hex prefix from all hex values
  return await rweb3.searchLogs(fromBlock, toBlock, addresses, topics, contractMetadata, true);
}
```

### call(methodName, params)
Executes a `callcontract`
```
// callcontract on a method named 'bettingEndBlock'
async function exampleCall(args) {
  const {
    senderAddress, // address
  } = args;

  return await contract.call('bettingEndBlock', {
    methodArgs: [],
    senderAddress: senderAddress,
  });
}
```

### send(methodName, params)
Executes a `sendtocontract`
```
// sendtocontract on a method named 'setResult'
async function exampleSend(args) {
  const {
    resultIndex, // number
    senderAddress, // address
  } = args;

  return await contract.send('setResult', {
    methodArgs: [resultIndex],
    gasLimit: 1000000, // setting the gas limit to 1 million
    senderAddress: senderAddress,
  });
}
```

## Encoder
`Encoder` static functions are exposed in Rweb3 instances.
```
const { Rweb3 } = require('rweb3');

const rweb3 = new Rweb3('http://bodhi:bodhi@localhost:13889');
rweb3.encoder.objToHash(abiObj, isFunction);
rweb3.encoder.addressToHex(address);
rweb3.encoder.boolToHex(value);
rweb3.encoder.intToHex(num);
rweb3.encoder.uintToHex(num);
rweb3.encoder.stringToHex(string, maxCharLen);
rweb3.encoder.stringArrayToHex(strArray, numOfItems);
rweb3.encoder.padHexString(hexStr);
rweb3.encoder.constructData(abi, methodName, args);
```

## Decoder
`Decoder` static functions are exposed in Rweb3 instances.
```
const { Rweb3 } = require('rweb3');

const rweb3 = new Rweb3('http://bodhi:bodhi@localhost:13889');
rweb3.decoder.toRunebaseAddress(hexAddress, isMainnet);
rweb3.decoder.removeHexPrefix(value);
rweb3.decoder.decodeSearchLog(rawOutput, contractMetadata, removeHexPrefix);
rweb3.decoder.decodeCall(rawOutput, contractABI, methodName, removeHexPrefix);
```

## Utils
`Utils` static functions are exposed in Rweb3 instances.
```
const { Rweb3 } = require('rweb3');

const rweb3 = new Rweb3('http://bodhi:bodhi@localhost:13889');
rweb3.utils.paramsCheck(methodName, params, required, validators);
rweb3.utils.appendHexPrefix(value);
rweb3.utils.trimHexPrefix(str);
rweb3.utils.chunkString(str, length);
rweb3.utils.toUtf8(hex);
rweb3.utils.fromUtf8(str);
rweb3.utils.isJson(str);
rweb3.utils.isRunebaseAddress(address);
```
